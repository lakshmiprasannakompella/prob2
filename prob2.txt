#include <bits/stdc++.h>
using namespace std;
class Solution
{
    public:
    int maxConsecutiveOnes(int N)
    {
          int curr=0,max=0;
        while(N>0){
            if((N&1)==1)
            curr++;
            else
            curr=0;
            if(max<curr)
            max=curr;
            N=N>>1;
        }
        return max;
    
    }
};
int main() {
	int t;
	cin>>t;//testcases
	while(t--)
	{
		int n;
		cin>>n;//input n
		Solution obj;
		//calling maxConsecutiveOnes() function
		cout<<obj.maxConsecutiveOnes(n)<<endl;
	}
	return 0;
}
